403     /**
404     * Construct a new Snake object.
405     *
406     * @class This is the class for the Snake
407     * @constructor
408     * @param board {Board} The board
409     * @param fruit {Fruit} The fruit
410     * @return {this} Returns the Snake
411     */
412     function Snake(board, fruit){
413         /** @private */ var lives = 3;
414         /** @private */ var posX = 6;
415         /** @private */ var posY = 13;
416         /** @private */ var auxX = 0;
417         /** @private */ var auxY = 0;
418         /** @private */ var tailX = 6;
419         /** @private */ var tailY = 13;
420         /** @private */ var arrayMoves = ['right']; //To save movements for tail moving later
421         /** @private */ var currentMove = arrayMoves[arrayMoves.length - 1];
422         /** @private */ var nextMove = 'right'; //To anticipate collisions        
423         /** @private */ var increaseTail = 3; //It will be 3 every time we eat a fruit
424         
425         board.setValue(posX, posY, 1); //To initialize the Snake on the board
426         
427         /**
428         * Get currentMove
429         *
430         * @return {String} Return the value
431         */
432 		this.getCurrentMove = function(){
433 			return currentMove;
434 		};
435 		
436 		/**
437         * Set score
438         *
439         * @param move {String} The next move to set
440         * @return {this} To be able to chain methods
441         */
442 		this.setNextMove = function(move){
443 			nextMove = move;
444 			return this;
445 		};
446 		
447 		/**
448         * Check the key pressed for the snake movement. As I have to
449         * use snake, I couldn't use it outside Snake "class" so here is it.
450         *
451         * @return {this} To be able to chain methods.
452         */
453 		this.checkKeys = function(){
454 		    if(Key.isDown(Key.UP)){
455                 if(currentMove !== 'down'){
456                     nextMove = 'up';
457                 }
458             }
459             if(Key.isDown(Key.DOWN)){
460                 if(currentMove !== 'up'){
461                     nextMove = 'down';
462                 }
463             }
464             if(Key.isDown(Key.LEFT)){
465                 if(currentMove !== 'right'){
466                     nextMove = 'left';
467                 }
468             }
469             if(Key.isDown(Key.RIGHT)){
470                 if(currentMove !== 'left'){
471                     nextMove = 'right';
472                 }
473             }
474             return this;
475 		};
476 		
477 		
478         /**
479         * Check if the next movement can be done.
480         * Catch the last movement on the array. Copy the coordinates on 
481         * auxiliary ones. Change the auxiliary coordinates.
482         * If that position has a 1 on the board check if it's a fruit or a wall.
483         * If it's a fruit or the position is 0, make the movement.
484         *
485         * @param board {Board} The board of the game to get values
486         * @param fruit {Fruit} The fruit, to check coordenates
487         * @return {Boolean} If the movement can be done, return true.
488         */
489         this.checkMove = function(board, fruit){
490             /** @private */ var move = arrayMoves[arrayMoves.length - 1];
491             auxX = posX;
492             auxY = posY;
493             if(move === 'up'){
494                 auxY--;
495                 currentMove = 'up';
496             }
497             else if(move === 'down')
498             {
499                 auxY++;
500                 currentMove = 'down';
501             }
502             else if(move === 'left'){
503                 auxX--;
504                 currentMove = 'left';
505             }
506             else if(move === 'right'){
507                 auxX++;
508                 currentMove = 'right';
509             }
510             if(board.getValue(auxX, auxY) === 1){ //A fruit or a wall
511                 if((auxX === fruit.getPosX()) && (auxY === fruit.getPosY())){ //A fruit
512                     this.eatFruit(board, fruit);
513                     return true;
514                 }
515                 else{ //A wall
516                     return false;
517                 }
518             }
519             else{ //A free move
520                 return true;
521             }
522         };
523         
524         /**
525         * Move the Snake.
526         * Check if the movement can be done.
527         * If true, copy the auxiliary coordinates to the real ones
528         * and push the nextMove to the arrayMoves.
529         *
530         * @param board {Board} The board of the game
531         * @param fruit {Fruit} The fruit
532         * @return {Boolean} To check if we have to move the Tail or not
533         */
534         this.moveSnake = function(board, fruit){
535             if(this.checkMove(board, fruit)){
536                 posX = auxX;
537                 posY = auxY;
538                 board.setValue(posX, posY, 1);
539                 arrayMoves.push(nextMove);
540                 return true;
541             }
542             else{
543                 lives--;
544                 if(lives === 0){
545                     gameOver = true;
546                     lives = 3;
547                 }
548                 board.initBoard();
549                 posX = 6;
550                 posY = 13;
551                 tailX = 6;
552                 tailY = 13;
553                 arrayMoves = ['right']; //To save movements for tail moving later
554                 currentMove = arrayMoves[arrayMoves.length - 1];
555                 nextMove = 'right'; //To anticipate collisions        
556                 increaseTail = 3; //It will be 3 every time we eat a fruit
557                 board.setValue(posX, posY, 1);
558                 //resetea fruta
559                 fruit.putNewFruit(board);
560                 return false;
561             }
562         };
563         
564         /**
565         * Move the tail of the Snake.
566         * If the tail is increasing, decrease it by 1.
567         * Else set the value of the tail on the array at 0.
568         * Catch the first movement on the arrayMoves.
569         * Change the coordinate.
570         *
571         * @param board {Board} The board to set values
572         * @return {this} To be able to chain functions
573         */
574         this.moveTail = function(board){
575             if(increaseTail === 0){
576                 board.setValue(tailX, tailY, 0);
577                 /** @private */ var move = arrayMoves.shift();
578                 if(move === 'up') tailY--;
579                 else if(move === 'down') tailY++;
580                 else if(move === 'left') tailX--;
581                 else if(move === 'right') tailX++;
582             }
583             else{
584                 increaseTail--;
585             }
586             return this;
587         };
588         
589         /**
590         * Eat the Fruit.
591         * Set increaseTail value to 3. Increase the board.score.
592         * Randomize the coordinates of the Fruit.
593         *
594         * @param board {Board} The board to set the score
595         * @param fruit {Fruit} The fruit to put a new one
596         * @return {this} To be able to chain functions
597         */
598         this.eatFruit = function(board, fruit){
599             increaseTail = 3;
600             board.setScore(board.getScore() + fruit.getScore());
601             fruit.putNewFruit(board);
602             return this;
603         };
604         return this;
605     };
